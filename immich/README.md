# Immich

```sh
IMMICH_VERSION=v2 # The Immich version to use. You can pin this to a specific version like "v2.1.0"
IMMICH_UPLOAD_LOCATION=./library # The location where your uploaded files are stored
IMMICH_PORT=2283 # Listening port
IMMICH_POSTGRES_VERSION=14-vectorchord0.4.3-pgvectors0.2.0@sha256:bcf63357191b76a916ae5eb93464d65c07511da41e3bf7a8416db519b40b1c23
IMMICH_POSTGRES_PASSWORD=postgres # Connection secret for postgres
IMMICH_POSTGRES_DATA_LOCATION=./postgres # The location where your database files are stored. Network shares are not supported for the database
IMMICH_POSTGRES_DB_STORAGE_TYPE=SSD # HDD, SSD
IMMICH_REDIS_VERSION=9@sha256:546304417feac0874c3dd576e0952c6bb8f06bb4093ea0c9ca303c73cf458f63
```
